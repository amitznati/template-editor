## Install

```bash
npm install --save template-editor
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'template-editor'
import 'template-editor/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
